-- Acessar o MySQL como root mysql -u root -p -- Criar o banco de dados caso não exista CREATE DATABASE IF NOT EXISTS jogos_bons; -- Selecionar o banco para uso USE jogos_bons; -- Criar tabela de categorias de jogos CREATE TABLE IF NOT EXISTS categoria ( id INT PRIMARY KEY AUTO_INCREMENT, -- identificador único da categoria nome VARCHAR(45) NOT NULL -- nome da categoria ); -- Criar tabela de jogos, vinculando à categoria CREATE TABLE IF NOT EXISTS jogo ( id INT PRIMARY KEY AUTO_INCREMENT, -- identificador único do jogo nome VARCHAR(45) NOT NULL, -- nome do jogo qtd INT NOT NULL, -- quantidade em estoque categoria_id INT NOT NULL, -- chave estrangeira da categoria FOREIGN KEY (categoria_id) REFERENCES categoria(id) -- vínculo com a tabela categoria ); -- Criar tabela de funcionários CREATE TABLE IF NOT EXISTS funcionario ( id INT PRIMARY KEY AUTO_INCREMENT, -- identificador único do funcionário nome VARCHAR(40) NOT NULL, -- nome do funcionário telefone VARCHAR(20) NOT NULL UNIQUE -- telefone único ); -- Criar tabela de clientes CREATE TABLE IF NOT EXISTS cliente ( id INT PRIMARY KEY AUTO_INCREMENT, -- identificador único do cliente nome VARCHAR(40) NOT NULL, -- nome do cliente cpf VARCHAR(20) NOT NULL UNIQUE -- CPF único do cliente ); -- Criar tabela de vendas relacionando cliente, jogo e funcionário CREATE TABLE IF NOT EXISTS venda ( id INT PRIMARY KEY AUTO_INCREMENT, -- identificador único da venda cliente_id INT NOT NULL, -- FK do cliente que comprou jogo_id INT NOT NULL, -- FK do jogo vendido funcionario_id INT NOT NULL, -- FK do funcionário responsável qtd INT DEFAULT 1, -- quantidade vendida FOREIGN KEY (cliente_id) REFERENCES cliente(id), FOREIGN KEY (jogo_id) REFERENCES jogo(id), FOREIGN KEY (funcionario_id) REFERENCES funcionario(id) );
